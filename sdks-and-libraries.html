Authentication Flow (JavaScript) 
import { Auth } from '@yourcompany/sdk-js';

// Initialize with your API key
const client = new Auth('YOUR_API_KEY');

try {
  // Get an authenticated user session
  const session = await client.login({ username, password });
  console.log('User ID:', session.userId);
} catch (error) {
  handleApiError(error); // Error handling pattern
}
Calling an Endpoint (Python) 
from yourcompany_sdk_py import Client

client = Client(api_key='YOUR_API_KEY')

# Call the data processing endpoint
response = client.data.process(
    source='stream', 
    params={'limit': 100}
)

for item in response.data:
    print(item['status'])
